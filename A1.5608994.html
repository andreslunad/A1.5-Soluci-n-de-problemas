# Cuartil uno
q1= data.Faltas.quantile(0.25)
# Cuartil tres
q3= data.Faltas.quantile(0.75)
# Rango intercuartil
iqr= q3-q1
# Lmite inferior
li= q1-(3*iqr)
# Lmite superior
ls= q3+(3*iqr)
# Observaciones fuera de cualquiera de los dos lmites
out= np.where((data.Faltas<=li) | (data.Faltas>=ls))[0]
# Imprimir informacin de outliers
print("Valores atpicos:")
